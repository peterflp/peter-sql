###

```sql
WITH random_customers AS (
    SELECT   customer_id
    FROM     foodie_fi.subscriptions
    ORDER BY RANDOM()
    LIMIT    8
)
SELECT   rc.customer_id,
         p.plan_id,
         p.plan_name,
         s.start_date
FROM     foodie_fi.plans p JOIN foodie_fi.subscriptions s USING(plan_id)
                           JOIN random_customers rc USING(customer_id)
ORDER BY 1, 2, 4
```
